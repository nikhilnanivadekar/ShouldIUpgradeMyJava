private String getQuarterNewSwitch(Integer monthNumber) {

    return switch (monthNumber) {
        case 1, 2, 3 -> "Q1";
        case 4, 5, 6 -> "Q2";
        case 7, 8, 9 -> "Q3";
        case 10, 11, 12 -> "Q4";
        default -> "Not a month in year";
    };
}
